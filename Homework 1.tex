    First, we will prove $\displaystyle{E(\sum_{i=1}^{n}X_i)=\sum_{i=1}^{n}(E(X_i))}$.\\
    \begin{proof}
        We will prove using induction.\\
        \underline{Base case}: For $n=1$, we have $E(X_1)=E(X_1)$. So, the base case holds.\\
        \underline{Inductive hypothesis}: Suppose that $\displaystyle{E(\sum_{i=1}^{n}X_i)=\sum_{i=1}^{n}(E(X_i))}$.
        Then,\\
        \ \\
        $\begin{aligned}[t]
        E(\sum_{i=1}^{n+1}X_i)&=E(\sum_{i=1}^{n}X_i+X_{n+1})\\
        &= E(\sum_{i=1}^{n}X_i)+E(X_{n+1})\\
        &=\sum_{i=1}^{n}E(X_i)+E(X_{n+1})\\
        &=\sum_{i=1}^{n+1}E(X_i).\\
        \end{aligned}$
        \ \\
        Thus, $\displaystyle{E(\sum_{i=1}^{n}X_i)=\sum_{i=1}^{n}(E(X_i))}$, by induction.\\
    \end{proof}

    Next, we will show that $\displaystyle{Cov(\sum_{i=1}^{n}X_i,Y)=\sum_{i=1}^{n}Cov(X_i,Y)}$.\\
    \ \\
    $\begin{aligned}[t]
        Cov(\sum_{i=1}^{n}X_i,Y)&=E((\sum_{i=1}^{n}X_i)Y)-E(\sum_{i=1}^{n}X_i)E(Y)\\
        &=E(\sum_{i=1}^{n}X_iY)-E(\sum_{i=1}^{n}X_i)E(Y)\\
        &=\sum_{i=1}^{n}E(X_iY)-(\sum_{i=1}^{n}E(X_i))E(Y)\\
        &=\sum_{i=1}^{n}E(X_iY)-\sum_{i=1}^{n}(E(X_i)E(Y))\\
        &=\sum_{i=1}^{n}[E(X_iY)-E(X_i)E(Y)]\\
        &=\sum_{i=1}^{n}Cov(X_i,Y)\\
    \end{aligned}$
